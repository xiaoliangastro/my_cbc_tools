240 def cal_pdf_property(x, pdf, conf_interval=90., hpd=False, diff=True, round_level=2, debug=False):
241  """Return lower (HPD)confidence bound, median, upper (HPD)confidence bound of a
242  function. x and pdf must have the same size. x do not need to be equally separated and
243  pdf do not need to be normalized. More pdf samples give more accurate properties.
244  """
245  from scipy.optimize import fmin
246 
247  def find_intersec(intersec, dx, pdf):
248  idx = pdf>intersec
249  prob = sum(pdf[idx]*dx[idx])
250  if debug:
251  print("Wanted prob: {}".format(conf_interval/100.))
252  print("\tGet: {}, pdf value: {}".format(prob, intersec[0]))
253  return (prob-conf_interval/100.)**2
254 
255  pdf = (pdf[:-1]+pdf[1:])/2.
256  dx = x[1:]-x[:-1]
257  pdf /= sum(dx*pdf)
258  cumsum_pdf = np.insert(np.cumsum(pdf*dx), 0, 0.0)
259  if debug:
260  print("normalized: {}".format(cumsum_pdf[-1]))
261  if hpd:
262  intersec = fmin(find_intersec, x0=max(pdf)/2., args=(dx, pdf), disp=False)
263  larger_than_intersec_idxes = np.where(pdf>intersec)[0]
264  lower_idx = larger_than_intersec_idxes[0]
265  upper_idx = larger_than_intersec_idxes[-1]
266  else:
267  conf_l = (100.-conf_interval)/2./100.
268  conf_u = (100.+conf_interval)/2./100.
269  lower_idx = np.where(cumsum_pdf>=conf_l)[0][0]
270  upper_idx = np.where(cumsum_pdf>=conf_u)[0][0]
271  median_idx = np.where(cumsum_pdf>=0.5)[0][0]
272  if debug:
273  print("intersection:{}".format(intersec))
274  print("idx_lower:{},idx_median:{},idx_upper:{}".format(lower_idx,median_idx,upper_idx))
275  print("lower:{},median:{},upper:{}".format(cumsum_pdf[lower_idx],cumsum_pdf[median_idx],cumsum_pdf[upper_idx]))
276  lower, median, upper = x[lower_idx], x[median_idx], x[upper_idx]
277  if diff:
278  lower = median-lower
279  upper = upper-median
280  return np.round([lower, median, upper], round_level)

478 def cal_kldiv(sample1, sample2, base=2., bw_method='scott', bins="rice", \
479  use_kde=False, cal_jsd=False):
480  """Calculate Kullback-Leibler divergence: KL(sample1|sample2)
481 
482  Parameters
483  ----------
484  sample1: 1D array like
485  sample1 to calculate
486  sample2: 1D array like
487  sample2 to calculate
488  base: float, optional
489  log base to calculate KL
490  bw_method: bw_method
491  see doc of scipy.stats.gaussian_kde
492  bins: bins
493  see doc of numpy.histogram
494  use_kde: bool, optional
495  use KDE method to mimic sample behavior or just use histogram method
496  cal_jsd: bool, optional
497  set to true to calculate Jensen-Shannon divergence
498 
499  Returns
500  -------
501  divergence: float
502  return JDS if cal_jsd else KLD
503  """
504 
505  from scipy.stats import gaussian_kde
506  from scipy.integrate import quad
507 
508  hist_ignore, bounds = np.histogram(np.append(sample1, sample2), bins=bins)
509  if use_kde:
510  lower_bds = bounds[:-1]
511  upper_bds = bounds[1:]
512  func1 = gaussian_kde(sample1, bw_method=bw_method)
513  func2 = gaussian_kde(sample2, bw_method=bw_method)
514  hist1 = [quad(func1, lb, ub)[0] for lb, ub in zip(lower_bds, upper_bds)]
515  hist2 = [quad(func2, lb, ub)[0] for lb, ub in zip(lower_bds, upper_bds)]
516  else:
517  hist1, _ = np.histogram(sample1, bounds)
518  hist2, _ = np.histogram(sample2, bounds)
519  hist1 = np.array(hist1)/float(sum(hist1))
520  hist2 = np.array(hist2)/float(sum(hist2))
521  if cal_jsd:
522  from scipy.spatial import distance
523  js_div = (distance.jensenshannon(hist1, hist2, base=base))**2
