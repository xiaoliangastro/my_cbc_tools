442 def cal_kldiv(sample1, sample2, base=2., bw_method='scott', scale_th=1e-8, \
443  bins="rice", use_kde=True, verbose=False, cal_jsd=False):
444  """Calculate Kullback-Leibler divergence: KL(sample1|sample2)
445 
446  Parameters
447  ----------
448  sample1: 1D array like
449  sample1 to calculate
450  sample2: 1D array like
451  sample2 to calculate
452  base: float, optional
453  log base to calculate KL
454  bw_method: bw_method
455  see doc of scipy.stats.gaussian_kde
456  scale_th: float, optional
457  bins with hist<scal_th/bin_num will be ignored to avoid bias
458  bins: bins
459  see doc of numpy.histogram
460  use_kde: bool, optional
461  use KDE method to mimic sample behavior or just use histogram method
462  verbose: bool, optional
463  set to true to show important results
464  cal_jsd: bool, optional
465  set to true to calculate Jensen-Shannon divergence
466 
467  Returns
468  -------
469  divergence: float
470  return JDS if cal_jsd else KLD
471  """
472 
473  from scipy.stats import gaussian_kde
474  from scipy.stats import entropy
475  from scipy.integrate import quad
476 
477  hist, bounds = np.histogram(sample1, bins=bins)
478  bin_num = len(hist)
479  if use_kde:
480  lower_bds = bounds[:-1]
481  upper_bds = bounds[1:]
482  func1 = gaussian_kde(sample1, bw_method=bw_method) #'silverman'
483  func2 = gaussian_kde(sample2, bw_method=bw_method) #'silverman'
484  norm_fac1 = quad(func1, min(sample1), max(sample1))[0]
485  norm_fac2 = quad(func2, min(sample2), max(sample2))[0]
486  if verbose:
487  print("Norm fac1:{}, norm fac2:{}".format(norm_fac1, norm_fac2))
488  normed_f1 = lambda x:func1(x)/norm_fac1
489  normed_f2 = lambda x:func2(x)/norm_fac2
490  hist1 = np.array([quad(normed_f1, lb, ub)[0] for lb, ub in \
491  zip(lower_bds, upper_bds)])
492  hist2 = np.array([quad(normed_f2, lb, ub)[0] for lb, ub in \
493  zip(lower_bds, upper_bds)])
494  if cal_jsd:
495  normed_f3 = lambda x: (normed_f1(x)+normed_f2(x))/2.0
496  hist3 = np.array([quad(normed_f3, lb, ub)[0] for lb, ub in \
497  zip(lower_bds, upper_bds)])
498  if verbose:
499  print("Hist1 sum:{}, hist2 sum:{}".format(sum(hist1), sum(hist2)))
500  else:
501  hist1 = hist
502  hist2, _ = np.histogram(sample2, bounds)
503  if cal_jsd: hist3 = (hist1+hist2)/2.0
504  #probability below threshold should be ignored to avoid unexpected bias
505  threshold = scale_th/bin_num
506  choose_index = (hist1>threshold)*(hist2>threshold)
507  if cal_jsd:
508  choose_index *= hist3>threshold
509  hist1 = hist1[choose_index]
510  hist2 = hist2[choose_index]
511  if cal_jsd:
512  hist3 = hist3[choose_index]
513  kld1 = entropy(hist1, qk=hist3, base=base)
514  kld2 = entropy(hist2, qk=hist3, base=base)
515  js_div = (kld1+kld2)/2.0
516  else:
517  kl_div = entropy(hist1, qk=hist2, base=base)
518  return js_div if cal_jsd else kl_div
